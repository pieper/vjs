'use strict';

var VJS = VJS || {};

VJS.extras.lut = VJS.extras.lut || {};

VJS.extras.lut.luts = function() {
  // we have 'none' and 'grayscale'
  // in order to compare it
  // it should be the same result
  return [
  'none',
  'spectrum',
  'hotandcold',
  'gold',
  'red',
  'green',
  'blue',
  'grayscale'
  ];
};

VJS.extras.lut.toIRGB = function(rawLUT) {
  var i = new Array(16);
  var r = new Array(16);
  var g = new Array(16);
  var b = new Array(16);
  for (var l = 0; l &lt; rawLUT.length; l++) {
    i[l] = rawLUT[l][0];
    r[l] = rawLUT[l][1];
    g[l] = rawLUT[l][2];
    b[l] = rawLUT[l][3];
  }

  return [i, r, g, b];
};

VJS.extras.lut.lut = function(label, data) {
  return {
    'label': label,
    'data': data
  };
};
